genes = load_reference_gene("data/refGene.txt.gz")
empirical_data = load_cnv_data("data/ISC-r1.CNV.bed")
def call_simulation(ref_gene, cnv_data, sim_args, causal_genes):
    i = 1
    for param1 in range(len(sim_args["prevalence"])):
        for param2 in range(len(sim_args["odds_ratio_params"]["shape"])):
            for param3 in range(len(sim_args['avg_cnv_per_individual'])):
                args_ins = {'block_size': sim_args['block_size'], 
                            "prevalence": sim_args['prevalence'][param1], 
                            "odds_ratio_params": {"shape": sim_args["odds_ratio_params"]["shape"][param2], 
                                                  "scale": sim_args["odds_ratio_params"]["scale"][param2]}, 
                            'avg_cnv_per_individual': sim_args['avg_cnv_per_individual'][param3], 
                            'n_case': sim_args["n_case"], 
                            'n_ctrl': sim_args["n_ctrl"]
                           }
                
                sample_del = simulate(ref_gene, pd.concat([cnv_data['delCases'], cnv_data['delControls']]), 
                                      args_ins, causal_genes["causal_genes_del"])
                sample_dup = simulate(ref_gene, pd.concat([cnv_data['dupCases'], cnv_data['dupControls']]), 
                                      args_ins, causal_genes["causal_genes_dup"])
                save_data(sample_dup, 'dup_sample_{}.pkl'.format(i))
                save_data(sample_del, 'del_sample_{}.pkl'.format(i))
                i += 1
    return sample_dup, sample_del
