# First toy: (1) Three CNVs overlap with two adjacent genes, only one of which is causal
#            (2) Three CNVs overlap with one causal gene, and the adjacent gene is noncausal
#            (3) Three CNVs overlap with one causal gene, one of which overlap with the adjacent noncausal gene
ref_gene = load_reference_gene("data/refGene.txt.gz")
ref_gene_name = ref_gene["gene_name"]
ref_cnv = load_cnv_data("data/ISC-r1.CNV.bed")
ref_cnv_data = pd.concat([ref_cnv['delCases'], ref_cnv['delControls']])
ref_cnv_length = ref_cnv_data['cnv_end'] - ref_cnv_data['cnv_start']
ref_cnv_mode = ref_cnv_length.mode().tolist()
toy_cnv_length = random.choice(ref_cnv_mode)
# get gene length
query = """
    SELECT chrom, gene_name, max(tx_end)-min(tx_start)
    FROM ref_gene
    GROUP BY chrom, gene_name
"""
ref_gene_table = sqldf(query)
ref_gene_table = ref_gene_table.rename(columns={"max(tx_end)-min(tx_start)": "gene_length"})
ref_gene_length = ref_gene_table["gene_length"].tolist()
ref_gene_median = np.median(ref_gene_length)
