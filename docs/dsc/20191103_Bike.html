import pandas as pd, numpy as np, matplotlib.pyplot as plt
from collections import Counter

data1 = pd.read_csv("/home/min/Downloads/Divvy_Trips/Divvy_Trips_2018_Q1.csv", sep = ",", header = 0)
data2 = pd.read_csv("/home/min/Downloads/Divvy_Trips/Divvy_Trips_2018_Q2.csv", sep = ",", header = 0)
data3 = pd.read_csv("/home/min/Downloads/Divvy_Trips/Divvy_Trips_2018_Q3.csv", sep = ",", header = 0)
data4 = pd.read_csv("/home/min/Downloads/Divvy_Trips/Divvy_Trips_2018_Q4.csv", sep = ",", header = 0)
data1.columns = list(data2.columns)

data = pd.concat([data1, data2, data3, data4])
data["tripduration"] = [float(x.replace(",", "")) for x in data["tripduration"]]
data = data[(~data["gender"].isna()) & (~data["birthyear"].isna())]
data["age"] = [2018-x for x in data["birthyear"]]
data = data[(data["age"] <= 80) & (data["tripduration"] < 10800)]
data["start"] = [int("{}{}".format(x.split(" ")[1].split(":")[0], x.split(" ")[1].split(":")[1])) for x in data["start_time"]]

toUchicago = data[data["to_station_id"].isin([328, 420, 423, 425, 426])]
fromUchicago = data[data["from_station_id"].isin([328, 420, 423, 425, 426])]
