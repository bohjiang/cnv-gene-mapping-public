import math
brad = [34.050536, -118.247861]
R_earth = 6371
def dist_to_brad(loc):
    loc = [float(s) for s in loc.strip("()").split(",")]
    return R_earth * math.sqrt(((loc[0] - brad[0])*math.pi/180)**2 + (math.cos((loc[0] + brad[0])/2 * math.pi/180) * (loc[1] - brad[1]) * math.pi/180)**2)
