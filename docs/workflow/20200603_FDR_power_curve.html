res = pd.DataFrame({"cutoff": cutoffs})
In [16]:
for x in [f"pip_{m1}", f"pip_{m2}", f"pip_{m3}", f"pip_{m4}", "p"]:
    res[f"{x}_FDR"] = [sum(pips[x]>=i)/n1 for i in cutoffs]
    res[f"{x}_power"] = [pips[(pips[x]>=i) & (pips["is_signal"] == 1)].shape[0]/n2 for i in cutoffs]
